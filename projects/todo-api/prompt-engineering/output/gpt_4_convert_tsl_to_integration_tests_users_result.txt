using Microsoft.AspNetCore.Mvc.Testing;
using System.Net;
using System.Net.Http.Json;
using System.Text.Json.Nodes;

namespace IntegrationTests

    public class UsersIntegrationTests : IClassFixture<WebApplicationFactory<Program>>
    {
        private readonly HttpClient _client;

        public UsersIntegrationTests(WebApplicationFactory<Program> factory)
        {
            _client = factory.CreateClient();
        }

        private Task<HttpResponseMessage> CreateUserAsync(string? username, string? password)
        {
            var request = new
            {
                username,
                password
            };

            return _client.PostAsJsonAsync("/users", request);
        }

        private Task<HttpResponseMessage> CreateTokenAsync(string? username, string? password)
        {
            var request = new
            {
                username,
                password
            };

            return _client.PostAsJsonAsync("/users/token", request);
        }

        [Fact]
        public async Task TC001_Create_User_When_Valid_Data_Returns_OK()
        {
            // arrange
            string username = $"user_{Guid.NewGuid():N}".Substring(0, 10);
            string password = "P@ssw0rd1";

            // act
            var response = await CreateUserAsync(username, password);

            // assert
            Assert.Equal(HttpStatusCode.OK, response.StatusCode);
        }

        public async Task TC002_Create_User_When_Username_Has_Minimum_Size_Returns_OK()

        [Fact]
        public async Task TC020_Authenticate_User_When_Valid_Data_Returns_OK()
        {
            // arrange
            string username = $"auth_{Guid.NewGuid():N}".Substring(0, 14);
            string password = "P@ssw0rd1";
            await CreateUserAsync(username, password);

            // act
            var response = await CreateTokenAsync(username, password);

            // assert
            var body = await response.Content.ReadFromJsonAsync<JsonObject>();
            var token = body?["token"]?.GetValue<string>();
            Assert.Equal(HttpStatusCode.OK, response.StatusCode);
            Assert.False(string.IsNullOrWhiteSpace(token));
        }

        [Fact]
