        private readonly HttpClient _client;

        public TodosReadIntegrationTests(WebApplicationFactory<Program> factory)
        {
            _client = factory.CreateClient();
        }

        private static string NewUsername(string prefix) => $"{prefix}_{Guid.NewGuid():N}";

        private Task<HttpResponseMessage> CreateUserAsync(string? username, string? password)
        {
            var payload = new Dictionary<string, object?>
            {
                ["username"] = username,
                ["password"] = password
            };

            return _client.PostAsJsonAsync("/users", payload);
        }

        private Task<HttpResponseMessage> CreateTokenAsync(string? username, string? password)
        {
            var payload = new Dictionary<string, object?>
            {
                ["username"] = username,
                ["password"] = password
            };

            return _client.PostAsJsonAsync("/users/token", payload);
        }

        private async Task<string> CreateUserAndGetTokenAsync(string username, string password)
        {
            var createUserResponse = await CreateUserAsync(username, password);
            Assert.Equal(HttpStatusCode.OK, createUserResponse.StatusCode);

            var tokenResponse = await CreateTokenAsync(username, password);
            Assert.Equal(HttpStatusCode.OK, tokenResponse.StatusCode);

            var tokenBody = await tokenResponse.Content.ReadFromJsonAsync<JsonObject>();
            Assert.NotNull(tokenBody);

            var token = tokenBody!["token"]!.GetValue<string>();
            Assert.False(string.IsNullOrWhiteSpace(token));

            return token;
        }

        private Task<HttpResponseMessage> CreateTodoAsync(string? token, string? title)
        {
            var payload = new Dictionary<string, object?>
            {
                ["title"] = title
            };

            var request = new HttpRequestMessage(HttpMethod.Post, "/todos")
            {
                Content = JsonContent.Create(payload)
            };

            if (!string.IsNullOrWhiteSpace(token))
            {
                request.Headers.Authorization = new AuthenticationHeaderValue("Bearer", token);
            }

            return _client.SendAsync(request);
        }

        private async Task<int> CreateTodoAndGetIdAsync(string token, string title)
        {
            var response = await CreateTodoAsync(token, title);
            Assert.Equal(HttpStatusCode.Created, response.StatusCode);

            var body = await response.Content.ReadFromJsonAsync<JsonObject>();
            Assert.NotNull(body);

            return body!["id"]!.GetValue<int>();
        }

        private Task<HttpResponseMessage> GetTodoAsync(string? token, int id)
        {
            var request = new HttpRequestMessage(HttpMethod.Get, $"/todos/{id}");

            if (!string.IsNullOrWhiteSpace(token))
            {
                request.Headers.Authorization = new AuthenticationHeaderValue("Bearer", token);
            }

            return _client.SendAsync(request);
        }

        private Task<HttpResponseMessage> ListTodosAsync(string? token)
        {
            var request = new HttpRequestMessage(HttpMethod.Get, "/todos");

            if (!string.IsNullOrWhiteSpace(token))
            {
                request.Headers.Authorization = new AuthenticationHeaderValue("Bearer", token);
            }

            return _client.SendAsync(request);
        }

        private Task<HttpResponseMessage> DeleteTodoAsync(string? token, int id)
        {
            var request = new HttpRequestMessage(HttpMethod.Delete, $"/todos/{id}");

            if (!string.IsNullOrWhiteSpace(token))
            {
                request.Headers.Authorization = new AuthenticationHeaderValue("Bearer", token);
            }

            return _client.SendAsync(request);
        }

        private static void AssertTodoArrayContainsSingleTodo(JsonArray? array, int expectedId, string expectedTitle, bool expectedIsComplete)
        {
            Assert.NotNull(array);
            Assert.Equal(1, array!.Count);

            var item = array[0] as JsonObject;
            Assert.NotNull(item);

            Assert.Equal(expectedId, item!["id"]!.GetValue<int>());
            Assert.Equal(expectedTitle, item["title"]!.GetValue<string>());
            Assert.Equal(expectedIsComplete, item["isComplete"]!.GetValue<bool>());
        }

        [Fact]
        public async Task TC035_List_Todos_When_User_Has_No_Todos_Returns_Empty_Array()
        {
            // arrange
            string username = NewUsername("list_empty");
            string password = "P@ssw0rd1";
            string token = await CreateUserAndGetTokenAsync(username, password);

            // act
            var response = await ListTodosAsync(token);

            // assert
            var body = await response.Content.ReadFromJsonAsync<JsonArray>();
            Assert.Equal(HttpStatusCode.OK, response.StatusCode);
            Assert.NotNull(body);
            Assert.Empty(body!);
        }

        [Fact]
        public async Task TC036_List_Todos_When_User_Has_Todos_Returns_Array()
        {
            // arrange
            string username = NewUsername("list_full");
            string password = "P@ssw0rd1";
            string token = await CreateUserAndGetTokenAsync(username, password);

            string title1 = "Todo 1";
            string title2 = "Todo 2";
            int todoId1 = await CreateTodoAndGetIdAsync(token, title1);
            int todoId2 = await CreateTodoAndGetIdAsync(token, title2);

            // act
            var response = await ListTodosAsync(token);

            // assert
            var body = await response.Content.ReadFromJsonAsync<JsonArray>();
            Assert.Equal(HttpStatusCode.OK, response.StatusCode);
            Assert.NotNull(body);
            Assert.Equal(2, body!.Count);

            bool foundTodo1 = false;
            bool foundTodo2 = false;

            foreach (var itemNode in body)
            {
                var item = itemNode as JsonObject;
                Assert.NotNull(item);

                var id = item!["id"]!.GetValue<int>();
                var title = item["title"]!.GetValue<string>();
                var isComplete = item["isComplete"]!.GetValue<bool>();

                Assert.True(id == todoId1 || id == todoId2);
                Assert.False(isComplete);

                if (id == todoId1)
                {
                    Assert.Equal(title1, title);
                    foundTodo1 = true;
                }

                if (id == todoId2)
                {
                    Assert.Equal(title2, title);
                    foundTodo2 = true;
                }
            }

            Assert.True(foundTodo1);
            Assert.True(foundTodo2);
        }

        [Fact]
        public async Task TC039_List_Todos_When_Using_Another_User_Token_Returns_Only_Own_Todos()
        {
            // arrange
            string ownerUsername = NewUsername("list_owner");
            string ownerPassword = "P@ssw0rd1";
            string ownerToken = await CreateUserAndGetTokenAsync(ownerUsername, ownerPassword);
            int ownerTodoId = await CreateTodoAndGetIdAsync(ownerToken, "Owner Todo");

            string otherUsername = NewUsername("list_other");
            string otherPassword = "P@ssw0rd1";
            string otherToken = await CreateUserAndGetTokenAsync(otherUsername, otherPassword);
            int otherTodoId = await CreateTodoAndGetIdAsync(otherToken, "Other Todo");

            // act
            var response = await ListTodosAsync(otherToken);

            // assert
            var body = await response.Content.ReadFromJsonAsync<JsonArray>();
            Assert.Equal(HttpStatusCode.OK, response.StatusCode);
            Assert.NotNull(body);
            Assert.Single(body!);

            var item = body[0] as JsonObject;
            Assert.NotNull(item);

            Assert.Equal(otherTodoId, item!["id"]!.GetValue<int>());
            Assert.Equal("Other Todo", item["title"]!.GetValue<string>());
            Assert.False(item["isComplete"]!.GetValue<bool>());
            Assert.NotEqual(ownerTodoId, item["id"]!.GetValue<int>());
        }
